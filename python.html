class Filedata:
def init(self):
self.b = 0
self.a = 0
self.e = 0
self.lp = 0
self.longestLine = 0
self.linenumber = 0
self.sumLength = 0
self.ave = 0


f = Filedata()

file = open('C:/test/dotlp_trace.txt', 'r')
f.b = 0
f.a = 0
f.e = 0
f.lp = 0
f.longestLine = 0
f.linenumber = 0
f.sumLength = 0
f.ave = 0

while True:
f.linenumber += 1

line = file.readline()
lengde = len(line)
f.sumLength += lengde

if lengde > f.longestLine:
f.longestLine = lengde

if not line:
break

if 'ERROR' in line:
# print(line)
f.e += 1
f.a += 1

elif 'lp_say' in line:
# print(line)
f.a += 1
f.lp += 1
f.b += 1


file.close()
f.ave = f.sumLength / f.b
print(f.b, " lines")
print(f.e, " Error")
print(f.lp, " lp")
print(f.a, " Error or lp")
print(f.longestLine, "Longest line")
print(f.linenumber, "number of lines")
print(f.ave, "ave")
